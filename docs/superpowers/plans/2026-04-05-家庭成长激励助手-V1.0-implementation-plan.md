// src/store/slices/authSlice.ts
import { StateCreator } from 'zustand'
import { UserMode } from '../../types'
import * as SecureStore from 'expo-secure-store'
import * as LocalAuthentication from 'expo-local-authentication'

export interface AuthSlice {
  mode: UserMode
  isAuthenticated: boolean
  passwordHash: string | null
  securityQuestion: string | null
  securityAnswerHash: string | null
  biometricsEnabled: boolean
  setMode: (mode: UserMode) => void
  authenticate: (password: string) => Promise<boolean>
  authenticateWithBiometrics: () => Promise<boolean>
  setPassword: (password: string) => Promise<void>
  setSecurityQuestion: (question: string, answer: string) => Promise<void>
  toggleBiometrics: (enabled: boolean) => void
}

export const createAuthSlice: StateCreator<AuthSlice> = (set, get) => ({
  mode: 'parent',
  isAuthenticated: false,
  passwordHash: null,
  securityQuestion: null,
  securityAnswerHash: null,
  biometricsEnabled: false,

  setMode: (mode) => set({ mode }),

  authenticate: async (password) => {
    // 密码验证逻辑
    const storedHash = await SecureStore.getItemAsync('parent_password')
    if (!storedHash) {
      // 首次使用，设置初始密码
      await SecureStore.setItemAsync('parent_password', password)
      set({ isAuthenticated: true, passwordHash: password })
      return true
    }
    // 实际项目使用bcrypt对比密码哈希
    const isValid = password === storedHash // 简化示例，实际用bcrypt.compare
    set({ isAuthenticated: isValid })
    return isValid
  },

  authenticateWithBiometrics: async () => {
    const result = await LocalAuthentication.authenticateAsync({
      promptMessage: '验证身份以进入家长模式',
      fallbackLabel: '使用密码',
    })
    if (result.success) {
      set({ isAuthenticated: true })
      return true
    }
    return false
  },

  setPassword: async (password) => {
    await SecureStore.setItemAsync('parent_password', password)
    set({ passwordHash: password })
  },

  setSecurityQuestion: async (question, answer) => {
    await SecureStore.setItemAsync('security_question', question)
    await SecureStore.setItemAsync('security_answer', answer)
    set({ securityQuestion: question, securityAnswerHash: answer })
  },

  toggleBiometrics: (enabled) => set({ biometricsEnabled: enabled }),
})
```

// src/store/slices/userSlice.ts
import { StateCreator } from 'zustand'
import { Child } from '../../types'
import userService from '../../services/db/userService'

export interface UserSlice {
  currentChild: Child | null
  children: Child[]
  loadChildren: () => Promise<void>
  setCurrentChild: (child: Child) => void
  addChild: (child: Omit<Child, 'id' | 'created_at'>) => Promise<void>
  updateChild: (id: number, data: Partial<Child>) => Promise<void>
  deleteChild: (id: number) => Promise<void>
}

export const createUserSlice: StateCreator<UserSlice> = (set, get) => ({
  currentChild: null,
  children: [],

  loadChildren: async () => {
    const children = await userService.getAll()
    set({ children, currentChild: children[0] || null })
  },

  setCurrentChild: (child) => set({ currentChild: child }),

  addChild: async (child) => {
    const id = await userService.create(child)
    const newChild = { ...child, id, created_at: new Date().toISOString() } as Child
    set(state => ({
      children: [...state.children, newChild],
      currentChild: state.currentChild || newChild
    }))
  },

  updateChild: async (id, data) => {
    await userService.update(id, data)
    set(state => ({
      children: state.children.map(c => c.id === id ? { ...c, ...data } : c),
      currentChild: state.currentChild?.id === id ? { ...state.currentChild, ...data } : state.currentChild
    }))
  },

  deleteChild: async (id) => {
    await userService.delete(id)
    set(state => ({
      children: state.children.filter(c => c.id !== id),
      currentChild: state.currentChild?.id === id ? state.children[0] || null : state.currentChild
    }))
  },
})
```
